# Terminal Feature Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the broken terminal on the Fun Stuff page with a polished 90×25 terminal that auto-runs neofetch with the Arch ASCII logo and x256-style color coding.

**Architecture:** Pure TypeScript + CSS — no new dependencies. `terminal.ts` is a full rewrite with neofetch output, a command registry, and an input handler. CSS updates scope all terminal styles to `.terminal` and add a small color-class set. Source Code Pro is loaded via Google Fonts using the existing preload pattern.

**Tech Stack:** TypeScript (strict, ESNext), plain CSS, Google Fonts (Source Code Pro)

## Task 3: Rewrite terminal.ts

**Files:**
- Full rewrite: `js/terminal.ts`

- [ ] **Step 1: Write the complete terminal.ts**

Replace the entire contents of `js/terminal.ts` with:

```typescript
const input = document.getElementById('commandInput') as HTMLInputElement;
const output = document.getElementById('output') as HTMLElement;
const funStuff = document.getElementById('fun-stuff') as HTMLElement;

function escapeHtml(str: string): string {
  return str
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;');
}

function neofetchOutput(): string {
  const logo: string[] = [
    '                   -`',
    '                  .o+`',
    '                 `ooo/',
    '                `+oooo:',
    '               `+oooooo:',
    '               -+oooooo+:',
    '             `/:-:++oooo+:',
    '            `/++++/+++++++:',
    '           `/++++++++++++++:',
    '          `/+++ooooooooooooo/`',
    '         ./ooosssso++osssssso+`',
    '        .oossssso-````/ossssss+`',
    '       -osssssso.      :ssssssso.',
    '      :osssssss/        osssso+++.',
    '     /ossssssss/        +ssssooo/-',
    '   `/ossssso+/:-        -:/+osssso+-',
    '  `+sso+:-`                 `.-/+oso:',
    ' `++:.                           `-/+/',
    '.`                                 `/',
  ];

  // Edit these values directly to match your real system info
  const info: string[] = [
    `<span class="c-bold">arch2kx</span><span class="c-dim">@</span><span class="c-bold">archlinux</span>`,
    `<span class="c-dim">-------------------</span>`,
    `<span class="c-arch">OS:</span>     <span class="c-val">Arch Linux x86_64</span>`,
    `<span class="c-arch">Kernel:</span> <span class="c-val">6.x.x-arch1-1</span>`,
    `<span class="c-arch">Uptime:</span> <span class="c-val">Xh Xm</span>`,
    `<span class="c-arch">Shell:</span>  <span class="c-val">zsh</span>`,
    `<span class="c-arch">WM:</span>     <span class="c-val">Plasma 6</span>`,
    `<span class="c-arch">CPU:</span>    <span class="c-val">[your cpu]</span>`,
    `<span class="c-arch">GPU:</span>    <span class="c-val">[your gpu]</span>`,
    `<span class="c-arch">Memory:</span> <span class="c-val">[used] / [total] MiB</span>`,
  ];

  const LOGO_WIDTH = 42;
  const lines: string[] = [];
  const total = Math.max(logo.length, info.length);

  for (let i = 0; i < total; i++) {
    const logoLine = (logo[i] ?? '').padEnd(LOGO_WIDTH);
    const infoLine = info[i] ?? '';
    lines.push(
      `<div><span class="c-arch">${logoLine}</span>  ${infoLine}</div>`
    );
  }

  return lines.join('');
}

const commands: Record<string, () => string> = {
  neofetch: () => neofetchOutput(),

  whoami: () => '<div>arch (arch2kx)</div>',

  help: () =>
    '<div>Available commands: <span class="c-arch">neofetch</span>, ' +
    '<span class="c-arch">whoami</span>, <span class="c-arch">projects</span>, ' +
    '<span class="c-arch">help</span>, <span class="c-arch">clear</span></div>',

  projects: () =>
    '<div>' +
    '<div>- <a href="https://arch2kx.github.io/aronasay/" class="c-arch" target="_blank" rel="noopener noreferrer">Aronasay</a> — cowsay but Arona from Blue Archive</div>' +
    '<div>- <a href="https://arch2kx.github.io/archs-ba-student-quiz/" class="c-arch" target="_blank" rel="noopener noreferrer">Blue Archive Student Quiz</a> — test your BA student knowledge</div>' +
    '</div>',

  clear: () => '',
};

function appendOutput(html: string): void {
  output.innerHTML += html;
  output.scrollTop = output.scrollHeight;
}

function handleCommand(raw: string): void {
  const cmd = raw.trim();
  if (cmd === '') return;

  appendOutput(
    `<div><span class="c-arch">arch2kx@archlinux ~ %</span> ${escapeHtml(cmd)}</div>`
  );

  if (cmd === 'clear') {
    output.innerHTML = '';
    return;
  }

  if (cmd === 'sudo rm -rf /') {
    appendOutput('<div><span class="c-err">zsh: permission denied: bro</span></div>');
    return;
  }

  const fn = commands[cmd];
  if (fn !== undefined) {
    appendOutput(fn());
  } else {
    appendOutput(
      `<div><span class="c-err">zsh: command not found: ${escapeHtml(cmd)}</span></div>`
    );
  }
}

input.addEventListener('keydown', (e: KeyboardEvent) => {
  if (e.key !== 'Enter') return;
  const value = input.value;
  input.value = '';
  handleCommand(value);
});

document.querySelector('.terminal')?.addEventListener('click', () => {
  input.focus();
});

// Focus input when Fun Stuff page becomes active
const observer = new MutationObserver(() => {
  if (funStuff.classList.contains('active')) {
    input.focus();
  }
});
observer.observe(funStuff, { attributes: true, attributeFilter: ['class'] });

// Auto-run neofetch on load
appendOutput(neofetchOutput());
```
